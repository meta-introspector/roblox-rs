Getting Started

After bundling your dependencies, here's how to use them in your Roblox game:

-- Require the roblox-rs package
local RobloxRs = require(game.ReplicatedStorage.Packages.RobloxRs)

-- Use your Rust dependencies
local serde = RobloxRs.serde
local result = serde.deserialize(someData)

-- Handle result using Rust-like patterns
if result.ok then
    local data = result:unwrap()
    print("Successfully deserialized:", data)
else
    local err = result.error
    warn("Deserialization error:", err)
end
